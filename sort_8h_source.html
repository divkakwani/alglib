35 /* Sorts the elements in [first, last) using selection sort
36  * @first The starting iterator of the range
37  * @last The one-past-the-end iterator of the range.
38  */
39 template<typename ForwardIter, typename BinaryPred>
40 void selection_sort(ForwardIter first, ForwardIter last, const BinaryPred& LT) {
41  for (auto it = first; it != last; it++)
42  std::swap(*it, *std::min_element(it, last, LT));
43 
44  /* Analysis
45  * Proof of Correctness:
46  * We prove it using the following invariant:
47  * Prior to an iteration of the loop, A[first..it-1] contains
48  * the smallest (it-first) elements of A in a non-decreasing order.
49  * Initialization: Initially, it = first, so A[first..first-1] contains
50  * no element. Hence it is trivially sorted.
51  * Maintenance: A[first..it-1] is sorted prior to an iteration, and in
52  * the loop A[it] is set to a minimum element in A[it..last].
53  * A[it] >= E, E ϵ A[first..it-1] because prior to the iteration,
54  * A[it..last] did not contain any element greater than E,
55  * E ϵ A[first..it-1]. Hence, A[first..it] contains its elements in a
56  * non-decreasing order. Further, the range also contains the smallest
57  * (it-first+1) elemnts of A since one more is added in the loop.
58  * Termination: At the end, we have, A[first..last] contains all the
59  * the elemets in a non-decreasing order. This proves the claim.
60  *
61  * Stable: Yes (Depends on the implementation of min_element. If it
62  * returns the first minimum element, then it is stable.)
63  * In-place: Yes
64  * Time-Complexity Gurantee: O(n^2).
65  * The loop runs n times and each of the iteration invoke min_element,
66  * which takes O(i) time where i goes from 1 to n.
67  * Mutual-Exclusion: required.
68  */
69 }

78 // Quick sort helper function
79 template<typename RandomAccessIter, typename BinaryPred>
80 RandomAccessIter randomized_partition(RandomAccessIter first,
81  RandomAccessIter last,
82  const BinaryPred& LT) {
83  /* The function selects a random element in [first, last) as pivot and
84  * partitions the interval around it.
85  * The part
86  */
87  int sz = std::distance(first, last);
88 
89  RandomAccessIter it = first + rand() % sz; // O(n) cost
90  std::swap(*it, *(last - 1));
91  auto pivot = last - 1;
92 
93  RandomAccessIter sep = first; // sep heads the larger-elements partition
94  for (auto it = first; it != last; it++) {
95  if (*it < *pivot) {
96  std::swap(*it, *sep);
97  sep++;
98  }
99  }
100  std::swap(*pivot, *sep);
101  return sep;
102 }
