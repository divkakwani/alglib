64  /* `node` is wrapper around the key with an additional pointer to

104 template<typename elt_t, typename key_t>
105 template<typename InputIter>
106 binary_heap<elt_t, key_t>::binary_heap(InputIter elt_first,
107  InputIter elt_last,
108  InputIter key_first) {
109  __sz = 0;
110  while (elt_first != elt_last) {
111  auto loc = make_entry(*elt_first);
112  heaparr[__sz++] = node_t {*key_first, loc};
113  ++elt_first;
114  }
115 
116  // Heapify the __stash
117  for (int i = __sz / 2; i >= 0; i--)
118  __sift_down(i);
119 }

162 void binary_heap<elt_t, key_t>::update_key(const elt_t& elt,
163  const key_t& key) {
164  if (nodeof.find(elt) == nodeof.end())
165  throw std::domain_error("The element doesn't exist");
166  int index_in_heap = nodeof[elt];
167  heaparr[index_in_heap] = node_t {key, heaparr[index_in_heap].elt_it};
168  __sift_up(index_in_heap);

172 template<typename elt_t, typename key_t>
173 void binary_heap<elt_t, key_t>::__sift_up(int node) {
174  /*
175  * Store the node in a temp and move all the other nodes encountered
176  * while going upwards down and finally put the temp in the upmost place.
177  */
178  node_t temp = heaparr[node];
179  while (node != 0 && temp < heaparr[__parent(node)]) {
180  heaparr[node] = heaparr[__parent(node)];
181  update_entry(node);
182  node = __parent(node);
183  }
184  heaparr[node] = temp;
185  update_entry(node);
186 }

188 template<typename elt_t, typename key_t>
189 void binary_heap<elt_t, key_t>::__sift_down(int node) {
190  int min_node = node;
191  int left_child = __left_child(node);
192  int right_child = __right_child(node);
193 
194  if (left_child < __sz && heaparr[left_child] < heaparr[min_node])
195  min_node = left_child;
196  if (right_child < __sz && heaparr[right_child] < heaparr[min_node])
197  min_node = right_child;
198 
199  if (min_node != node) {
200  std::swap(heaparr[node], heaparr[min_node]);
201  update_entry(node);
202  update_entry(min_node);
203  __sift_down(min_node);
204  }
