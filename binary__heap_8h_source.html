70  /* `nodeof` stores the node corresponding to element inserted in the heap.

102 template<typename elt_t, typename key_t>
103 template<typename InputIter>
104 binary_heap<elt_t, key_t>::binary_heap(InputIter elt_first,
105  InputIter elt_last,
106  InputIter key_first) {
107  __sz = 0;
108  while (elt_first != elt_last) {
109  auto loc = make_entry(*elt_first);
110  heaparr[__sz++] = node_t {*key_first, loc};
111  ++elt_first;
112  }
113 
114  // Heapify the __stash
115  for (int i = __sz / 2; i >= 0; i--)
116  __sift_down(i);
117 }

160 void binary_heap<elt_t, key_t>::update_key(const elt_t& elt,
161  const key_t& key) {
162  if (nodeof.find(elt) == nodeof.end())
163  throw "Non-existent element";
164  int index_in_heap = nodeof[elt];
165  heaparr[index_in_heap] = node_t {key, heaparr[index_in_heap].elt_it};

170 template<typename elt_t, typename key_t>
171 void binary_heap<elt_t, key_t>::__sift_up(int node) {
172  /*
173  * Store the node in a temp and move all the other nodes encountered
174  * while going upwards down and finally put the temp in the upmost place.
175  */
176  node_t temp = heaparr[node];
177  while (node != 0 && temp < heaparr[__parent(node)]) {
178  heaparr[node] = heaparr[__parent(node)];
179  update_entry(node);
180  node = __parent(node);
181  }
182  heaparr[node] = temp;
183  update_entry(node);
184 }

186 template<typename elt_t, typename key_t>
187 void binary_heap<elt_t, key_t>::__sift_down(int node) {
188  int min_node = node;
189  int left_child = __left_child(node);
190  int right_child = __right_child(node);
191 
192  if (left_child < __sz && heaparr[left_child] < heaparr[min_node])
193  min_node = left_child;
194  if (right_child < __sz && heaparr[right_child] < heaparr[min_node])
195  min_node = right_child;
196 
197  if (min_node != node) {
198  std::swap(heaparr[node], heaparr[min_node]);
199  update_entry(node);
200  update_entry(min_node);
201  __sift_down(min_node);
202  }
203 }
